[documenti]def parsing_varesion(data: str):
    """Elabora la versione specificata target speciale $interpolation

    Setta la versione del parser

    Args:
        data(str): valore della versione

    Returns:
        void

    Raises:
        Exception: se non è una bool
    """
    if not isinstance(data, str):
        raise Exception("Target $version is not str")
    core.global_settings.setenv("PARSER_VERSION_DEFAULT", data)
    return data

    """Elabora i dati presenti nel target speciale $interpolation

    Setta l'interpolazione dei valori a False o True (che di default è True)

    Args:
        data(bool): valore del target $interpolation

    Returns:

    """
    if not isinstance(data, bool):

    core.global_settings.setenv("TEMPLATE_TARGET_INTERPOLATION", data)
    return data




[documenti]def parsing_env(data: dict) -> None:
    """Elabora i dati presenti nel target speciale $env

    Se target_data una stringa verrà elaborata come path di un file ini.
    Se invece è un dict allora viene caricato come chiave valore

    Args:
        data(dict): valore del target $env

    Returns:
        void

    Raises:
        Exception: se non è una str o un dict
    """
    if not isinstance(data, dict):
        raise Exception("Target '%s' is not dict or str" % str(data))
    for (key, value) in data.items():
        if core.global_settings.getenv("TEMPLATE_TARGET_INTERPOLATION") == "True":
            value = core.text.interpolation_vars(value)
        core.global_settings.load_from_keyvalue(key=key, value=value, overwrite=True)
        core.io.info("ENV -> %s=%s" % (key, value))




[documenti]def rendering_stages(stages: dict, global_stage: dict = {}) -> dict:
    """Esegue il rendering degli stages, ogni stage è composto in uno dei seguenti modi:

        - dict, chiave valore, in cui la chiave è il nome dello stage e il valore è un \
            dict contenete la struttura standard dello stage
        - str, chiave valore, in cui la chiave è il nome dello stage e il valore è una \
            stringa che si riferisce al nome di uno stage globale, in questo caso verrà copiata \
                lo stage globale all'interno degli stages

    Args:
        stages (dict): valore del file yaml processato
        global_stage (dict): valore dei target protetti

    Returns:
        dict
    """

    def command(module: str, args="") -> str:
        cmd = "python3 -m %s " % module

        if isinstance(args, str):
            return cmd + args

        if isinstance(args, dict):
            return cmd + " ".join(
                [
                    str(opt) + (" " + str(value) if value else "")
                    for (opt, value) in args.items()
                ]
            )

        if isinstance(args, list):
            return cmd + " ".join([str(value) for value in args])

        raise Exception("Argument 'args' not valid, str o list.")

    parsed = {}
    for (name, value) in stages.items():
        # recupera lo stage globale
        if isinstance(value, str):
            if value not in global_stage.keys():
                core.io.critical("Global stage not found: %s" % value)
            parsed[name] = global_stage.get(value).copy()
            continue

        parsed[name] = {
            "command": command(value["module"], value.get("args", "")),
            "before": value.get("before", {}),
            "after": value.get("after", {}),
        }
    return parsed




[documenti]def parsing_protected_target(data: dict) -> dict:
    """Controlla e processa i target protetti presenti nel template

    Se riscontra le parole chiave dei target speciali allora esegue la funzione relativa.

    Args:
        data(dict): valore del file yaml processato

    Returns:
        dict
    """
    parsed = {
        "version": core.global_settings.getenv("PARSER_VERSION_DEFAULT"),
        "interpolation": core.global_settings.getenv("TEMPLATE_TARGET_INTERPOLATION"),
        "env": {},
        "global_stage": {},
    }

    if "$version" in data.keys():
        parsed["version"] = parsing_varesion(str(data.get("$version")))

    if "$interpolation" in data.keys():
        parsed["interpolation"] = parsing_interpolation(data.get("$interpolation"))

    if "$env" in data.keys():
        parsed["env"] = parsing_env(data.get("$env"))

    if "$global_stage" in data.keys():
        parsed["global_stage"] = rendering_stages(data.get("$global_stage"))

    return parsed




[documenti]def parsing_target(name: str, data: dict, protected: dict) -> dict:
    """Esegue il parsing del target nello specifico :

        1. Stage -> esegue il rendering degli stage inglobando anche quelli globali
        2. Options -> controlla e recupera le opzioni
        3. Before e After -> controlla e recupera le azione before e after
        4. Env -> elabora l'env definito nel target

    Args:
        name (dict): nome del target
        data (dict): valore del file yaml processato
        protected (dict): valore dei target protetti

    Returns:
        dict
    """
    target = core.global_settings.getenv("TEMPLATE_FILE_STRUCTURE_TARGET").copy()
    if "stage" in data.keys():
        if not isinstance(data, dict):
            raise Exception(
                "Target options 'stage' is not valid format. (required dict)"
            )
        target["stage"] = rendering_stages(
            data.get("stage"), protected.get("global_stage", {})
        )
    else:
        raise Exception("Target not found 'stage'")

    if "options" in data.keys():
        for opt in target["options"]:
            if opt not in data["options"].keys():
                continue
            if target["options"][opt].keys() == data["options"][opt].keys():
                target["options"][opt] = data["options"][opt]
                break
            core.io.critical("Target %s option: %s not valid" % (name, opt))

    if "after" in data.keys():
        if isinstance(data.get("after"), dict):
            target["after"] = data.get("after")
        else:
            core.io.critical("Target %s 'after' section not valid" % name)

    if "before" in data.keys():
        if isinstance(data.get("before"), dict):
            target["before"] = data.get("before")
        else:
            core.io.critical("Target %s 'before' section not valid" % name)

    if "env" in data.keys():
        env = {}
        if core.global_settings.getenv("TEMPLATE_TARGET_INTERPOLATION") == "True":
            for (key, value) in data.get("env").items():
                env[key] = core.text.interpolation_vars(
                    value, core.global_settings.environ(**env)
                )
        else:
            env = data.get("env")

        target["env"] = env

    return target




[documenti]def load(data: dict) -> dict:
    """Processa i dati estrapolati dal template eseguendo le sostituzioni, le interpolazioni, \
        la creazione del comando e dell'ambiente.

    Args:
        data(dict): valori contenuti nel file yaml processato

    Returns:
        dict
    """
    try:
        parsed = core.global_settings.getenv("TEMPLATE_FILE_STRUCTURE")
        parsed["origin"] = data.copy()

        # elaborazione target protetti
        parsed_protected = parsing_protected_target(data)
        # rimuove i target protetti dai target
        for protected in core.global_settings.getenv("TEMPLATE_TARGET_PROTECTED"):
            parsed[protected.replace("$", "")] = parsed_protected.get(
                protected.replace("$", "")
            )
            if protected in data.keys():
                data.pop(protected)

        # elaborazione target
        for (name, target) in data.items():
            parsed["target"][name] = parsing_target(name, target, parsed_protected)

        return parsed
    except Exception as e:
        core.io.exception(e)
