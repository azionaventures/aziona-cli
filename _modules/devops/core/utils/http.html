import requests

from devops.core import core



[documenti]def scan_response(response) -> None:
    try:
        core.io.debug(response.content.decode())
        response.raise_for_status()
    except requests.exceptions.HTTPError as e:
        core.io.exception(e, "HTTP error occurred")
    except Exception as e:
        core.io.exception(e)




[documenti]def post(
    url: str,
    data: dict,
    headers: dict = {},
    check_response: bool = True,
    scan_response_func=None,
    **kargs
):
    import json

    data = json.dumps(data)

    response = requests.post(url, headers=headers, data=data, **kargs)

    if check_response is True:
        if callable(scan_response_func):
            scan_response_func(response)
        else:
            scan_response(response)

    return response
