"""Il modulo global_setting.py gestisce le configurazioni globali per i moduli devops.

Le configurazioni base sono sovrascrivibili in due modi:

- **Variabili d'ambiente** con cui è possibile sovrascrivere le configurazioni globali esempio se si volesse sovrascrivere il \
    path del file log (che in devops è definito in LOGGING_PATH) LOGGING_PATH=/example/path nel proprio ambiente.
- **Run-time** usando la funzione *setenv* presente in global_setting oppure usando il modulo devops.core.loader.

"""
import os
from datetime import datetime

EXTENDS = {}
"""
Le variabili che vengono caricate nell'env attraverso caricate in run-time dalla funzione 'setenv'.
Consente di overridire le configurazioni globali e di tenere traccia delle varibili aggiunte dopo l'init.
"""

SESSION = {}



[documenti]def load_from_keyvalue(
    key: str,
    value: str,
    overwrite: bool = False,
    with_suffix: str = None,
    from_session: bool = False,
) -> None:
    """Caricamento di una nuova variabile di ambiente

    Args:
        key (str): Nome variabili d'ambiente
        value (str): Valore
        overwrite (bool,optional): Overwrite variabile d'ambiente se esiste già
        with_suffix (str,optional): Suffisso che da aggiungere alla 'key'

    Returns:
        None
    """
    if key in os.environ.keys() and overwrite is False:
        return
    index = with_suffix + key if isinstance(with_suffix, str) else key
    if from_session is True:
        setsession(key=index, value=value)
        return

    setenv(key=index, value=value)




[documenti]def load_from_dict(
    overwrite: bool = False,
    with_suffix: str = None,
    from_session: bool = False,
    **kargs
) -> None:
    """Caricamento varibili di ambiente a partire da un dict.

    Se il valore è un dict verrà caricato con la chiave principale seguita dalla secondaria, KEY_KEYCHILD=VALUECHILD.
    Ex. {test:{subtest:'ok'}} => TEST_SUBTEST=ok

    Args:
        overwrite (bool,optional): Overwrite variabile d'ambiente se esiste già
        with_suffix (str,optional): Suffisso che da aggiungere alla 'key'
        **kargs: chiave=valore

    Returns:
        None
    """
    for key, value in kargs.items():
        if isinstance(kargs[key], dict):
            [
                load_from_keyvalue(key + "_" + k, v, overwrite, with_suffix)
                for k, v in kargs[key].items()
            ]
        else:
            load_from_keyvalue(key, value, overwrite, with_suffix, from_session)




[documenti]def setsession(key: str, value: str):
    if not isinstance(key, str):
        raise Exception("ENVIRON KEY is not str")

    if not isinstance(value, str):
        value = str(value)

    key = key.upper()
    globals()["SESSION"].update({key: value})
    os.environ[key] = value

[documenti]def getenv(key: str, default=None, only_os_environ: bool = False) -> str:
    """Recupera le variabili d'ambiente settate, controllando in sequenza:

        1. Dalla variabile EXTENDS contenente le var settate in run-time
        2. Dalla configurazione global_settings
        3. Dall'os.environ

        In caso only_os_environ a True esegue direttamente il punto 3

    Args:
        key (str): nome variabile da recuperare
        default (): valore da ritornare in caso di nessuna corrispondenza per la chiave richiesta
        host_env (bool,optional): recupera i valori solo dal os.environ

    Returns:
        mixed: ritorna la stringa della variabile richiesta, oppure il valore di deafault indicato
    """
    if only_os_environ is False:
        if key in EXTENDS.keys():
            return EXTENDS[key]

        if key in globals().keys():
            return globals()[key]

    return os.getenv(key, default)




[documenti]def environ(**kargs: str) -> dict:
    """Restituisce l'ambiente dell'intero processo run-time, unendo:

        1. os.environ
        2. global_settings (solo ed esclusivamente i valore stringa)
        3. SESSION
        4. EXTENDS
        5. gli argomenti passati alla funzione

        Vengono mergiati in un unico dict (in modo crescente vengono sovrascritte le chiavi uguali)

    Args:
        **kargs(str): valori da aggiungere all'environ
    Returns:
        dict: environ completo
    """
    # Copia l'environ
    environ = os.environ.copy()

    # Copia le variabili stringe del modulo
    for key, value in globals().copy().items():
        if isinstance(value, str) and not (key.startswith("__") or key.endswith("__")):
            environ[key] = value

    # Copia i valori di EXTENDS
    environ.update(**EXTENDS)

    # Copia i valori degli argomenti
    for key, value in kargs.items():
        if isinstance(value, str):
            environ[key] = value

    return environ
