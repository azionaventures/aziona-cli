# -*- coding: utf-8 -*-
"""Il modulo è l'entrypoint per il devosps-runner. \
Ha il compito di processare il file yml contentente il template con le azioni da eseguire. \
Ogni azione viene eseguita come sotto processo e inizializzato con un env costruito a partire
da quello host più eventuali varibili definite nel template. \
L'esecuzione può essere di uno o più target in modo sequenziale (in ordine di inserimento da command line), se
l'esecuzione va in errore allora il processo master si interromperà non eseguendo i successivi target indicati,
è possibile inibire l'interruzione per uno o più target con apposite regole nel template.
"""

import argparse
import os
import sys

from devops.core import core

[documenti]def parse_target(target: dict) -> tuple:
    """Esegue il parser del target, interpolando le varibili se attivo, e costruisce \
        la varibile contenente le varibili d'ambiente da passare ai sotto processi del target \
        costruita con i valori della macchina host + quelli iniettati dal template ($env e la chiave env del target)

    Args:
        target(dict): il target contenuto nel template yaml

    Returns:
        tuple: il primo valore è il target e il secondo l'env
    """
    env = core.global_settings.environ(**target.get("env", {}))
    if core.global_settings.getenv("TEMPLATE_TARGET_INTERPOLATION") == "True":
        # interpolzaione degli stage
        for (name, value) in target["stage"].items():
            target["stage"][name]["command"] = core.text.interpolation_vars(
                value["command"], from_dict=env
            )
            target["stage"][name]["after"] = core.text.interpolation_vars(
                value["after"], from_dict=env
            )
            target["stage"][name]["before"] = core.text.interpolation_vars(
                value["before"], from_dict=env
            )

        # interpolzaione dei comandi before del target
        if "before" in target.keys():
            for (name, value) in target["before"].items():
                target["before"][name] = core.text.interpolation_vars(
                    value, from_dict=env
                )

        # interpolzaione dei comandi after del target
        if "after" in target.keys():
            for (name, value) in target["after"].items():
                target["after"][name] = core.text.interpolation_vars(
                    value, from_dict=env
                )

    return (target, env)




[documenti]def exec_before_or_after(data: dict, env: dict, allow_failure=False) -> None:
    """Esegue i comandi definiti nell'argomento data

    Args:
        data(dict): ogni chiave del dict avrà come valore un comando che verrà eseguito
        env(dict,optional): environment da passare al sotto processo
        allow_failure(list,bool,optional): consente di non eseguire l'eccezzione se settato a True\
            oppure se viene passato una lista di stringhe e il 'name' è presente nella suddetta lista

    Returns:
        None:
    """
    for (name, cmd) in data.items():
        if isinstance(allow_failure, list):
            allow_failure = name in allow_failure
        if not isinstance(allow_failure, bool):
            core.io.critical("Argument 'allow_failure' not boolean")
        try:
            core.commands.exec(cmd, env=env)
        except Exception as e:
            message = "Esecuzione comando fallita:\n- %s: %s" % (name, cmd)
            if allow_failure is True:
                core.io.warning(message + " (allow_failure 'attivo')")
            else:
                core.io.exception(e, message)




[documenti]def exec_stage(name: str, cmd: str, env: dict = {}, allow_failure=False) -> None:
    """Esegue il modulo definito nello stage.

    Args:
        name(str): nome dello stage eseguito
        cmd(str): comando eseguito come sotto processo
        env(dict,optional): environment da passare al sotto processo
        allow_failure(list,bool,optional): consente di non eseguire l'eccezzione se settato a True\
            oppure se viene passato una lista di stringhe e il 'name' è presente nella suddetta lista

    Returns:
        None:
    """
    try:
        # controllo se il nome dello stage è presente nella lista di accettazione
        if isinstance(allow_failure, list):
            allow_failure = name in allow_failure
        if not isinstance(allow_failure, bool):
            core.io.critical("Argument 'allow_failure' not boolean")

        if not bool(env):
            env = os.environ.copy()

        core.commands.exec_check(cmd, env=env)
    except Exception as e:
        message = "Esecuzione stage '%s' fallita " % name
        if allow_failure is True:
            core.io.warning(message + "- allow_failure 'attivo'")
        else:
            core.io.exception(e, message)




[documenti]def load(args) -> None:
    """Esegue la logica del modulo.

    Args:
        args(dict,argparse.Namespace,optional): argomenti richiesti dal modulo

    Returns:
        None:
    """
    try:
        if isinstance(args, dict):
            args = argparse.Namespace(**args)

        if not isinstance(args, argparse.Namespace):
            core.io.critical("Argomenti non validi")

        template = core.parser.load(filepath=args.file)

        for target_name in args.targets:
            if target_name not in template["target"]:
                core.io.warning("Target %s non trovato nel template" % target_name)
                continue

            target, env = parse_target(template["target"][target_name])

            # esegue gli script definiti nel campo 'before'  del target
            exec_before_or_after(
                data=target.get("before", {}),
                env=env,
                allow_failure=target["options"]["allow_failure"]["before"],
            )

            # esegue gli stage (before, stage command e in fine after) del target
            for (stage_name, stage_data) in target["stage"].items():
                core.io.info("Stage: %s" % stage_name)
                exec_before_or_after(
                    data=stage_data.get("before", {}),
                    env=env,
                    allow_failure=target["options"]["allow_failure"]["stage"],
                )
                exec_stage(
                    name=stage_name,
                    cmd=stage_data["command"],
                    env=env,
                    allow_failure=target["options"]["allow_failure"]["stage"],
                )
                exec_before_or_after(
                    data=stage_data.get("after", {}),
                    env=env,
                    allow_failure=target["options"]["allow_failure"]["stage"],
                )

            # esegue gli script definiti nel campo 'after'  del target
            exec_before_or_after(
                data=target.get("after", {}),
                env=env,
                allow_failure=target["options"]["allow_failure"]["after"],
            )
    except Exception as e:
        core.io.exception(e)
